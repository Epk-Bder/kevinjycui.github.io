import java.io.*; import java.util.*; public class linesweep { static class layer implements Comparable<layer> { int height; int direction; public layer(int h, int d) { height = h; direction = d; } public int compareTo(layer o) { return height-o.height; } } public static void main(String[] args) { Scanner sc = new Scanner(System.in); int n = sc.nextInt(); int h = sc.nextInt(); layer [] obstacles = new layer [200001]; int [] line = new int [500001]; int count = 0; for (int i=0; i<n; i+=2) { obstacles[i] = new layer(h-sc.nextInt(), 1); line[obstacles[i].height]++; obstacles[i+1] = new layer(sc.nextInt(), 2); line[obstacles[i+1].height]--; count++; } Arrays.sort(obstacles, 0, n-1); int num = 0; int min = 200001; for (int j=0; j<h; j++) { count+=line[j]; if (count<min) { min=count; num=0; } if (count==min) { num++; } } System.out.println(min+" "+num); System.exit(0); } }
